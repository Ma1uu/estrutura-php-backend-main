DELIMITER $$
 
CREATE PROCEDURE salvar_categoria (
    IN var_id INT,
    IN var_nome VARCHAR(100),
    IN var_informacoes TEXT
)
BEGIN
 
    IF var_id IS NULL OR var_id = 0 THEN
 
        INSERT INTO categoria (nome, informacoes)
        VALUES (var_nome, var_informacoes);
 
    ELSE
 
        UPDATE categoria
        SET
            nome = var_nome,
            informacoes = var_informacoes
        WHERE id = var_id;
 
    END IF;
 
END$$
 
DELIMITER ;
